select emptb.name,emptb.Posit,emptb.FileNo,emptb.EducationQualification, emptb.Yearofpassing,emptb.ExperienceInThisOrg,emptb.ExperienceOutside,emptb.DateOfJoining, salarybonus.PresentSalary,salarybonus.IncrementAmt,salarybonus.DateOfIssue, grade.MgtApprovedGrade,grade.Remarks,grade.DateOfIssue from emptb,salarybonus,grade where salarybonus.DateOfIssue=(select max(salarybonus.DateOfIssue) from salarybonus,emptb where emptb.FileNo=salarybonus.FileNo) and grade.DateOfIssue=(select max(grade.DateOfIssue) from grade,emptb where emptb.FileNo=grade.FileNo)
